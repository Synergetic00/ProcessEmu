package programs;

import javafx.scene.canvas.*;
import javafx.scene.paint.*;
import javafx.scene.shape.*;

import utils.PVector;
import static utils.PVector.*;
import java.io.*;
import static utils.FXUtils.*;
import main.FXApp;

public class CodeGenTest extends FXApp {

    int rad = 60;         // Width of the shape
    double xpos, ypos;    // Starting position of shape    

    double xspeed = 2.8;  // Speed of the shape
    double yspeed = 2.2;  // Speed of the shape

    int xdirection = 1;   // Left or Right
    int ydirection = 1;   // Top to Bottom

    public CodeGenTest(GraphicsContext gc) {
        super(gc);
    }

    public void setup() {
        size(640, 360);
        noStroke();
        ellipseMode(RADIUS);
        // Set the starting position of the shape
        xpos = width/2;
        ypos = height/2;
    }

    public void draw() {
        background(102);
  
        xpos = xpos + ( xspeed * xdirection );
        ypos = ypos + ( yspeed * ydirection );

        if (xpos > width-rad || xpos < rad) {
            xdirection *= -1;
        }

        if (ypos > height-rad || ypos < rad) {
            ydirection *= -1;
        }

        ellipse(xpos, ypos, rad, rad);
    }
}
